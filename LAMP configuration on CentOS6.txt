Icinga Installation steps

#Pre-req install LAMP

#stop iptables
service iptables stop
chkconfig iptables off

#selinux disable
set disable in selinux config file

#install ntp server
yum install ntp -y
service ntpd start
chkconfig ntpd on


#Install LAMP on centos 6

#Install httpd server
yum install httpd
chkconfig httpd on
service httpd start
#validate httpd Installation from web browser


#install MySQL
#install wget
yum install wget
wget http://repo.mysql.com/mysql-community-release-el6-7.noarch.rpm
rpm -ivh mysql-community-release-el6-7.noarch.rpm
yum install mysql-server
service mysqld start
chkconfig mysqld on
service mysqld restart
service mysqld status

#Reset MySQL password
#For fresh install MySQL pwd is blank

#login to MySQL and reset the password
mysql -u root
mysql> use mysql;
mysql> update user set password=PASSWORD("Use your password here- My password is password") where user='root';
mysql> flush privileges;
mysql> quit


# Install PHP

yum install php php-mysql php-gd php-pear -y
